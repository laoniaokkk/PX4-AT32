		s32k3xx_autoleds.c

		s32k3xx_buttons.c
		s32k3xx_clockconfig.c
		i2c.cpp
		init.c
		mtd.cpp
		s32k3xx_periphclocks.c
		spi.cpp
		timer_config.cpp
		s32k3xx_userleds.c
	)

	target_link_libraries(drivers_board
		PRIVATE
			nuttx_arch # sdio
			nuttx_drivers # sdio
			drivers__led # drv_led_start
			px4_layer
			arch_io_pins
		)
